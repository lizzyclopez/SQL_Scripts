IF (@@ERROR <> 0 OR @ReturnCode <> 0) GOTO QuitWithRollback

DECLARE @jobId BINARY(16)
EXEC @ReturnCode =  msdb.dbo.sp_add_job @job_name=N'00:00 IPACS_Load_Process 2 (On Demand)', 
		@enabled=0, 
		@notify_level_eventlog=0, 
		@notify_level_email=0, 
		@notify_level_netsend=0, 
		@notify_level_page=0, 
		@delete_level=0, 
		@description=N'Steps:

		@category_name=N'[Uncategorized (Local)]', 
		@owner_login_name=N'sa', @job_id = @jobId OUTPUT
IF (@@ERROR <> 0 OR @ReturnCode <> 0) GOTO QuitWithRollback
/****** Object:  Step [Run VMUsageMonthly]    Script Date: 9/12/2018 11:21:00 AM ******/
EXEC @ReturnCode = msdb.dbo.sp_add_jobstep @job_id=@jobId, @step_name=N'Run VMUsageMonthly', 
		@step_id=1, 
		@cmdexec_success_code=0, 
		@on_success_action=1, 
		@on_success_step_id=0, 
		@on_fail_action=2, 
		@on_fail_step_id=0, 
		@retry_attempts=0, 
		@retry_interval=0, 
		@os_run_priority=0, @subsystem=N'TSQL', 
		@command=N'DECLARE @BillingMonth as varchar(10), @profile_name varchar(255), @retval int

--Set billing month to be passed in as parameter for the stored procedure.
set @BillingMonth = convert(varchar, DATEADD(dd,-(DAY(DATEADD(mm,1,getdate()))-1),DATEADD(mm,-1,getdate())),101)

--Execute stored procedure
EXEC @retval = ComputeServices..VMUsageMonthly @BillingMonth

if @retval = 0 -- success
	exec msdb.dbo.sp_send_dbmail @profile_name = ''SQL_DBMail'', @recipients = ''lizzy.lopez@hpe.com'', @subject = ''CRS 132C IPACS Load - VM Usage Succeessful'', @body = ''The stored procedure VMUsageMonthly ran successfully!''
else
	exec msdb.dbo.sp_send_dbmail @profile_name = ''SQL_DBMail'', @recipients = ''lizzy.lopez@hpe.com'', @subject = ''CRS 132C IPACS Load - VM Usage Failed!'', @body = ''The stored procedure VMUsageMonthly failed to run!''
', 
		@database_name=N'ComputeServices', 
		@flags=0
IF (@@ERROR <> 0 OR @ReturnCode <> 0) GOTO QuitWithRollback
